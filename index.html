Discord Bot (Demo)

/check-user (discordId)

Dieser Bot ist eine Demo-Version für Kunden: Er simuliert einen Check eines Discord-Users gegen interne Risiko-Signale (z.B. bekannte Cheat-Communities, Re-Join Pattern, Link-Cluster). In einer echten Version werden die Datenquellen/Listen/Regeln individuell für dein Netzwerk angebunden.

Was der Bot kann

01

Slash Command: /check-user <discordId> mit strukturierter Ausgabe (Risk Score, Flags, Hinweise).

02

Server-/Community-Risiko-Matching (Demo: simuliert). Real: Matching über gepflegte Watchlists + Heuristiken.

03

Audit-Logs & Export (z.B. JSON/Webhook). Real: Integrationen in Admin-Panel/CRM möglich.

// Demo: discord.js v14 (vereinfachtes Beispiel) // /check-user <discordid> - liefert Risiko-Signale (simuliert) import { Client, GatewayIntentBits, SlashCommandBuilder } from "discord.js"; const client = new Client({ intents: [GatewayIntentBits.Guilds] }); const command = new SlashCommandBuilder() .setName("check-user") .setDescription("Check a Discord user against risk signals") .addStringOption(opt => opt.setName("discordid").setDescription("Discord ID").setRequired(true)); function simulateCheck(discordId) { const profileSeed = discordId.split("").reduce((a,c)=> (a + (c.charCodeAt(0) * 17)) % 997, 0) const flagged = (profileSeed % 100) < 37 const source = demoRiskDb["Analystic Network"] const base = flagged ? 62 : 8 const variance = (profileSeed % 29) const score = Math.min(99, base + variance) const flags = [] if(flagged){ if((profileSeed % 2) === 0) flags.push("watchlist_match") if((profileSeed % 3) === 0) flags.push("invite_cluster") if((profileSeed % 5) === 0) flags.push("alt_pattern") if((profileSeed % 7) === 0) flags.push("link_spam") } else { flags.push("no_match") } const seenServers = flagged ? source.servers.slice(0, 1 + (profileSeed % source.servers.length)) : [] return { discordId, score, flags, seenServers, provider: "Analystic Network (Demo)" } } function renderBotDemo(result){ const out = document.getElementById("botDemoOut") const line = document.createElement("div") const levelClass = result.score >= 70 ? "bad" : (result.score >= 40 ? "warn" : "ok") line.className = "demoLine" line.innerHTML = `
[BOT] /check-user ${result.discordId}
Provider: ${result.provider}
Risk Score: ${result.score}/100
Flags: ${result.flags.join(", ")}
Related Servers: ${result.seenServers.length ? result.seenServers.join(", ") : "—"}
` out.prepend(line) while(out.children.length>25) out.removeChild(out.lastChild) } function runBotDemo(discordId){ const out = document.getElementById("botDemoOut") if(!isValidDiscordId(discordId)){ const line = document.createElement("div") line.className = "demoLine" line.innerHTML = `
